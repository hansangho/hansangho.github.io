arguments, callee

  1. arguments : 함수에 전달된 파라미터를 배열과 같은 형식으로 가지고 있는 객체
  2. arguments.callee : 실행된 함수 자신

parameter 확장

  1. 파라미터 갯수가 많고 명확해야 할 경우 Object 로 전달하면 좋다.
  2. jQuery.extend 함수의 활용 

// none jQuery
function paramTest(_option){
    if(!_option.name) _option.name = 'hansangho';
    if(!_option.age) _option.age = 39;

    console.log(  JSON.stringify(_option) );
}
paramTest({ name: 'angra', age: 30 });
 
// use jQuery
function paramJqueryTest(_option){
    var defaultOption = {
        name: 'hansangho',
        age: 39
    };
    jQuery.extend(true, defaultOption, _option);
    console.log(  JSON.stringify(defaultOption) );
}
paramJqueryTest({ name: 'bora', age: 29 });

고계함수, 익명함수

  • 고계함수: 인수로 전달되는 함수
  • 익명함수: 인수로 전달 할 경우 익명함수를 많이 이용한다.

var sangho = {
    name: 'sangho',
    cry: '잉잉',
    shout: '멍 멍'
};
var angra = {
    name: 'angra',
    cry: 'ang ang',
    shout: 'mung mung'
};
function humanSound(data, paramfunc) {
    paramfunc( data );
}
function cry ( data ) {
    console.log(data.name, '님이 웁니다.',data.cry);
}
function shout( data ) {
    console.log(data.name, '님이 소리칩니다.',data.shout);
}
humanSound(sangho, cry);
humanSound(angra, shout);
humanSound(angra, cry);

클로저


function showName(firstName, lastName) {
    var nameIntro = "Your name is ";
    return function () {
        console.log(nameIntro + firstName + " " + lastName);
    }
}
var nameshow = showName("Michael", "Jackson");
nameshow();
 
// Your name is Michael Jackson

호이스팅(Hoisting)

  • 변수 선언을 최상단으로 올리는 자바스크립트의 기본 동작
  • 자바스크립트에서 선언은 최상위로 이동된다.



var x = 5;
console.log(x); // 5
console.log(y); // undefined
var y = 7;


//위 코드가 아래처럼 된다는 얘기
var x;
var y;
 
x = 5;
console.log(x); // 5
console.log(y); // undefined
y = 7;
